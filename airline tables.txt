create table Aircraft_type( A_ID int primary key, Capacity int, A_model varchar(50), A_weight int, Company varchar(50));

create table Route( Route_ID int primary key, Departure time, Destination VARCHAR(50), R_type VARCHAR(20));

CREATE TABLE Flight (Flight_ID INT PRIMARY KEY, Departure TIME,Arrival TIME,Flight_date DATE,A_ID INT,FOREIGN KEY (A_ID) REFERENCES Aircraft_type (A_ID));

CREATE TABLE Airfare (Fare_ID INT PRIMARY KEY,Class varchar(50),Charge_amount decimal(8,2),Description varchar(50),Flight_ID int,FOREIGN KEY (Flight_ID) REFERENCES Flight (Flight_ID));

create table Passengers(P_ID int primary key, P_name varchar(50),DOB date, Street varchar(50),House_no int, City varchar(50),Contact int,Sex enum('Male','Female'),Flight_ID int, FOREIGN KEY (Flight_ID) REFERENCES Flight (Flight_ID));

create table Transaction(Ts_ID int primary key, Ts_type enum('Booking', 'Ticket Purchase', 'Ticket Cancellation', 'Ticket Change', 'Refund'),Departure_date date,Booking_date date,Charged_amount decimal(8,2), P_ID int, Flight_ID int,Fare_ID int,  FOREIGN KEY (Flight_ID) REFERENCES Flight (Flight_ID),FOREIGN KEY (Fare_ID) REFERENCES Airfare (Fare_ID),FOREIGN KEY (P_ID) REFERENCES Passengers(P_ID));

create table Countries(Country_code int primary key, Country_name varchar(50));

create table Airport( Air_code int primary key, Name varchar(50),State varchar(50),City Varchar(50),Country_code int, Foreign key (Country_code) references Countries(Country_code));

create table can_land( Air_code int, Flight_ID int,FOREIGN KEY (Flight_ID) REFERENCES Flight (Flight_ID),FOREIGN KEY (Air_code) REFERENCES Airport(Air_code));

create table travels_on(Route_ID int, Flight_ID int,FOREIGN KEY (Flight_ID) REFERENCES Flight (Flight_ID),foreign key (Route_ID) references Route(Route_ID));

create table Passenger_contact(P_ID int, Contact int,primary key(P_ID,Contact),foreign key (P_ID) REFERENCES Passengers(P_ID));
